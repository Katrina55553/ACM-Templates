
**置换环**

对于一个排列 0 ~ n - 1 将其排序的最小交换次数为
`idx` 与 `val` 建边，每个环的长度减一之和，即 **n - 图中环的数量**

---

https://www.luogu.com.cn/problem/P12176

```c++
void solve() {
    int n; cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        a[i]--;
    }
    int cnt = 0;
    vector<int> idx(n, -1);
    for (int i = 0; i < n; ++i) {
        if (idx[i] == -1) {
            int cur = i;
            while (idx[cur] == -1) {
                idx[cur] = cnt;
                cur = a[cur];
            }
            cnt++;
        }
    }
    int ans = n - cnt;
    cout << ans << "\n";
}
```
